let sparkles = []; function setup() { createCanvas(600, 400); background(255); cursor(); // Default system cursor textFont('Courier New'); // Use Courier New } function draw() { background(255); // Generate sparkles frequently if (random() < 0.4) { sparkles.push(new Sparkle(mouseX, mouseY)); } for (let i = sparkles.length - 1; i >= 0; i--) { sparkles[i].update(); sparkles[i].show(); if (sparkles[i].finished()) { sparkles.splice(i, 1); } } } class Sparkle { constructor(x, y) { this.x = x + random(-3, 3); this.y = y + random(-3, 3); this.alpha = 255; this.size = random(10, 14); // Slightly smaller = appears thinner this.letter = String.fromCharCode(random(97, 123)); // lowercase a-z } update() { this.alpha -= 1; } finished() { return this.alpha <= 0; } show() { noStroke(); // no outline = looks thinner fill(0, this.alpha); // only fill the text textSize(this.size); textAlign(CENTER, CENTER); text(this.letter, this.x, this.y); } }
